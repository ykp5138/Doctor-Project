# SETUP — requires Python 3.10 exactly (whisperx won't work on 3.11+)
#
# 1. Install Python 3.10 via pyenv:
#      brew install pyenv
#      pyenv install 3.10.14 && pyenv local 3.10.14
#
# 2. Create + activate venv:
#      python -m venv venv && source venv/bin/activate
#
# 3. Install PyTorch FIRST (before this file):
#      pip install torch==2.1.0 torchaudio==2.1.0
#
# 4. Then install this file:
#      pip install -r requirements.txt
#
# 5. System deps (brew):
#      brew install ffmpeg
#      brew install ollama && ollama pull llama3
#
# 6. Create .env in project root:
#      HF_TOKEN=<huggingface token with pyannote access>
#      ASSEMBLYAI_API_KEY=<assemblyai key>
#
# 7. Run:
#      Terminal 1: ollama serve
#      Terminal 2: uvicorn main:app --reload
#      Terminal 3: cd frontend && npm install && npm run dev
